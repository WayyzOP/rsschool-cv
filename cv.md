function getYear(year) {
    if (year >= 1987 && year <= 2012) {
        return '2013';
    }
    
    year += 1;
 
    var yearSorted = String(year).split('').sort();
 
    for (var i = 0; i < yearSorted.length - 1; i++) {
        if (yearSorted[i] == yearSorted[i + 1]) {
            return getYear(year);
        } else {
            if (i == yearSorted.length - 2) {
                return year;
            }
        }
    }
    
}
